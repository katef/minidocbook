	\xmlsetsetup{\xmldocument}{html|head|body}{xml:*}

	\xmlsetsetup{\xmldocument}{h1|h2|h3|h4}{xml:*}

	\xmlsetsetup{\xmldocument}{ol|ul|li}{xml:*}
	\xmlsetsetup{\xmldocument}{dl|dt|dd}{xml:*}

	\xmlsetsetup{\xmldocument}{p}{xml:*}
	\xmlsetsetup{\xmldocument}{span|div}{xml:*}
\stopxmlsetups
