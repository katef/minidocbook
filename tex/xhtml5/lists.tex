
\startxmlsetups xml:mdb-xhtml5-list
    \xmlsetsetup{\xmldocument}{ol|ul|li}{xml:*}
    \xmlsetsetup{\xmldocument}{dl|dt|dd}{xml:*}
\stopxmlsetups

\xmlregistersetup{xml:mdb-xhtml5-list}

\startxmlsetups xml:li
	\item
	\xmlflush{#1}
\stopxmlsetups

\startxmlsetups xml:ul
	\startitemize
	\xmlflush{#1}
	\stopitemize
\stopxmlsetups

\startxmlsetups xml:ol
	\startitemize[n]
	\xmlflush{#1}
	\stopitemize
\stopxmlsetups


\startxmlsetups xml:dl
	\par
	\xmlflush{#1}
\stopxmlsetups

\startxmlsetups xml:dt
	\xmlflush{#1}
	\crlf
\stopxmlsetups

\startxmlsetups xml:dd
	\startnarrower
	\xmlflush{#1}
	\stopnarrower
\stopxmlsetups
