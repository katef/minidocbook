
\startxmlsetups xhtml:setup-primitive
	\xmlsetsetup{\xmldocument}{i|b|em|tt|q}{xhtml:*}
	\xmlsetsetup{\xmldocument}{sub|sup}{xhtml:*}
	\xmlsetsetup{\xmldocument}{p}{xhtml:*}
	\xmlsetsetup{\xmldocument}{br}{xhtml:*}
\stopxmlsetups

\xmlregistersetup{xhtml:setup-primitive}

% not supported in html5
\startxmlsetups xhtml:tt
	{\tt \xmlflush{#1}}
\stopxmlsetups

\startxmlsetups xhtml:b
	{\bf \xmlflush{#1}}
\stopxmlsetups

\startxmlsetups xhtml:i
	{\it \xmlflush{#1}}
\stopxmlsetups

\startxmlsetups xhtml:em
	{\em \xmlflush{#1}}
\stopxmlsetups

\startxmlsetups xhtml:sub
	\low{\xmlflush{#1}}
\stopxmlsetups

\startxmlsetups xhtml:sup
	\high{\xmlflush{#1}}
\stopxmlsetups

\startxmlsetups xhtml:q
	\quotation{\xmlflush{#1}}
\stopxmlsetups

\startxmlsetups xhtml:p
	\xmlflush{#1}
	\par
\stopxmlsetups

\startxmlsetups xhtml:br
	\crlf
\stopxmlsetups
