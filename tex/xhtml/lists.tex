
\startxmlsetups xhtml:setup-list
	\xmlsetsetup{\xmldocument}{ol|ul|li}{xhtml:*}
	\xmlsetsetup{\xmldocument}{dl|dt|dd}{xhtml:*}
\stopxmlsetups

\xmlregistersetup{xhtml:setup-list}

\startxmlsetups xhtml:li
	\item
	\xmlflush{#1}
\stopxmlsetups

\startxmlsetups xhtml:ul
	\startitemize
	\xmlflush{#1}
	\stopitemize
\stopxmlsetups

\startxmlsetups xhtml:ol
	\startitemize[n]
	\xmlflush{#1}
	\stopitemize
\stopxmlsetups


\startxmlsetups xhtml:dl
	\par
	\xmlflush{#1}
\stopxmlsetups

\startxmlsetups xhtml:dt
	\xmlflush{#1}
	\crlf
\stopxmlsetups

\startxmlsetups xhtml:dd
	\startnarrower
	\xmlflush{#1}
	\stopnarrower
\stopxmlsetups
