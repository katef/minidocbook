
\startxmlsetups xml:table
	%\setupTABLE[c][1][align=right]
	%\setupTABLE[c][2][align=right]
	\bTABLE[frame=on,split=yes]
		\xmlflush{#1}
	\eTABLE
\stopxmlsetups

\startxmlsetups xml:thead
	\xmlflush{#1}
\stopxmlsetups

\startxmlsetups xml:tbody
	\xmlflush{#1}
\stopxmlsetups

\startxmlsetups xml:tr
	\bTR
	\xmlflush{#1}
	\eTR
\stopxmlsetups

\startxmlsetups xml:th
	% TODO: use <col/> for alignment
	\bTD[align=middle,style=bold]
	\xmlflush{#1}
	\eTD
\stopxmlsetups

\startxmlsetups xml:td
	\doifnumberelse {\xmlatt{#1}{rowspan}}
	{
		\bTD[nr=\xmlatt{#1}{rowspan},align=lohi]
		\xmlflush{#1}
		\eTD
	}
	{
		\bTD
		\xmlflush{#1}
		\eTD
	}
\stopxmlsetups
