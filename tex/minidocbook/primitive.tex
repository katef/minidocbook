
% Docbook elements which map directly to native ConTeXt commands.
% These should be all "presentational" Docbook elements, rather than
% anything with particularly meaningful semantics.

\startxmlsetups mdb:setup-primitive
	\xmlsetsetup{\xmldocument}{emphasis}{mdb:*}
	\xmlsetsetup{\xmldocument}{subscript|superscript}{mdb:*}
	\xmlsetsetup{\xmldocument}{quote}{mdb:*}
	\xmlsetsetup{\xmldocument}{para}{mdb:*}
\stopxmlsetups

\xmlregistersetup{mdb:setup-primitive}

\startxmlsetups mdb:emphasis
	{\em \xmlflush{#1}}
\stopxmlsetups

\startxmlsetups mdb:subscript
	\low{\xmlflush{#1}}
\stopxmlsetups

\startxmlsetups mdb:superscript
	\high{\xmlflush{#1}}
\stopxmlsetups

\startxmlsetups mdb:quote
	\quotation{\xmlflush{#1}}
\stopxmlsetups

\startxmlsetups mdb:para
	\xmlflush{#1}
	\par
\stopxmlsetups
